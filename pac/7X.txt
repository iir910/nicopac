
var fq='socks5 127.0.0.1:5050';
var zl='DIRECT';


var gg="*google*|*.gstatic.com"

var sns="twitter.com|*.facebook.com"

var du="*.hk|*.jp|*.tw"

function FindProxyForURL(url, host) {
    if (shExpMatch(host, gg)) {
        return fq;
    }

    if (shExpMatch(url, du)) {
        return fq;
    }

    if (shExpMatch(url, sns)) {
        return fq;
    }


    return zl ;
}
